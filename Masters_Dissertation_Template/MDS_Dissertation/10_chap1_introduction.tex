\begin{itemize}

    \item LexRank \citep{erkan-radev-2004-lexpagerank}: Similar to PageRank, it ranks sentences based on eigenvector centrality.
\end{itemize}

\subsection*{Example Techniques of Abstractive Summarization:}
\begin{itemize}
    \item Pointer-Generator Networks \citep{see-etal-2017-get}: Combines the benefits of copying words from the source text and generating new words.
    \item \gls{t5} : A transformer-based model pre-trained on multiple \gls{nlp} tasks including summarization .
    \item \gls{bart} \citep{lewis2020bart}: Denoising autoencoder for pretraining sequence-to-sequence models, especially effective for the abstractive summarization.
\end{itemize}
